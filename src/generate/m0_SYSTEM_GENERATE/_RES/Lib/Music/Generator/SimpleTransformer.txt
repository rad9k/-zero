"STORE"
	import @System\Lib\Music direct meta
	import @System\Lib\Std direct meta
	class "SimpleTransformer"
		attribute "NoteShift" @Integer <<"-11":"11">>
			default "0"
			<@$Section :: "Pitch">
		attribute "OctaveShift" @Integer <<"-10":"10">>
			default "0"
			<@$Section :: "Pitch">
		attribute "VelocityShift" @Integer <<"-127":"127">>
			default "0"
			<@$Section :: "Velocity">
		attribute "VelocityMin" @Integer <<"0":"127">>
			default "0"
			<@$Section :: "Velocity">
		attribute "VelocityMax" @Integer <<"0":"127">>
			default "127"
			<@$Section :: "Velocity">
		attribute "LengthShift" @Integer <<"-1536":"1536">>
			default "0"
			<@$Section :: "Length">
		attribute "LengthMultiply" @Float <<"0":"4">>
			default "1"
			<@$Section :: "Length">
		attribute "LengthMin" @Integer <<"0":"1536">>
			default "0"
			<@$Section :: "Length">
		attribute "LengthMax" @Integer <<"0":"1536">>
			default "1536"
			<@$Section :: "Length">
		attribute "ControlChangeShift" @Integer <<"-127":"127">>
			default "0"
			<@$Section :: "ControlChange">
		attribute "ControlChangeMin" @Integer <<"0":"127">>
			default "0"
			<@$Section :: "ControlChange">
		attribute "ControlChangeMax" @Integer <<"0":"127">>
			default "127"
			<@$Section :: "ControlChange">
		association "Input" @Sequence "0":"1"
			<@$Section :: "Sequences">
		association "Output" @Sequence "0":"1"
			<@$Section :: "Sequences">
		attribute "AdditionalOctaveShift" @Integer
		method "CopyAtomAttributes" ()
			<@System\Meta\Presentation\$Hide :: @$Empty>
			Output\ExtendTimeLength = %Input\ExtendTimeLength
			if Output\Length === ~00
				Output +< @@HasLength\Length :: %"0"
			Output\Length = %Input\Length
			if Output\IsDrum === ~00
				Output +< @@Sequence\IsDrum :: %"False"
			if Input\IsDrum<> == "1"
				Output\IsDrum = %Input\IsDrum
		method "DoNoteShift" (@Integer "note")
			variable "nNote" @Integer
			this\'AdditionalOctaveShift' = "0"
			nNote = note + this\'NoteShift'
			if nNote > "11"
				nNote = nNote - "12"
				this\'AdditionalOctaveShift' = "1"
			if nNote < "0"
				nNote = nNote + "12"
				this\'AdditionalOctaveShift' = "-1"
			return nNote
		method "DoOctaveShift" (@Integer "octave")
			variable "nOctave" @Integer
			nOctave = octave + this\'OctaveShift' + this\'AdditionalOctaveShift'
			if nOctave > "9"
				nOctave = "9"
			if nOctave < "-1"
				nOctave = "-1"
			return nOctave
		method "DoVelocityShift" (@Integer "velocity")
			variable "nVelocity" @Integer
			nVelocity = velocity + this\'VelocityShift'
			if nVelocity > this\VelocityMax
				nVelocity = this\VelocityMax
			if nVelocity < this\VelocityMin
				nVelocity = this\VelocityMin
			return nVelocity
		method "DoLengthShift" (@Integer "length")
			variable "nLength" @Integer
			nLength = length + this\'LengthShift'
			nLength = nLength * this\LengthMultiply
			if nLength > this\LengthMax
				nLength = this\LengthMax
			if nLength < this\LengthMin
				nLength = this\LengthMin
			return nLength
		method "DoControlChangeShift" (@Integer "CCValue")
			variable "nCCValue" @Integer
			nCCValue = CCValue + this\'ControlChangeShift'
			if nCCValue > this\ControlChangeMax
				nCCValue = this\ControlChangeMax
			if nCCValue < this\ControlChangeMin
				nCCValue = this\ControlChangeMin
			return nCCValue
		method "CopyEvents" ()
			<@System\Meta\Presentation\$Hide :: @$Empty>
			variable "note" @Integer
			variable "octave" @Integer
			variable "velocity" @Integer
			variable "length" @Integer
			variable "CCvalue" @Integer
			Output -< Output\Event:
			for vertex "e" in Input\Event:
				Output +< @@History\Event :: e{
					if e\$Is:NoteEvent<> == "1"
						note = this.'DoNoteShift'[e\Note]
						octave = this.'DoOctaveShift'[e\Octave]
						velocity = this.'DoVelocityShift'[e\Velocity]
						length = this.'DoLengthShift'[e\Length]
						@@$Is ::> @@NoteEvent
						@@Event\TriggerTime :: %e\TriggerTime
						@@HasLength\Length :: length
						@@Pitch\Octave :: octave
						@@Pitch\Note :: note
						@@Note\Velocity :: velocity
					if e\$Is:ControlChangeEvent<> == "1"
						CCvalue = this.'DoControlChangeShift'[e\Value]
						@@$Is ::> @@ControlChangeEvent
						@@Event\TriggerTime :: %e\TriggerTime
						@@ControlChange\Number :: %e\Number
						@@ControlChange\Value :: CCvalue
				}
		method "PutIntoStore" ()
			<@System\Meta\Presentation\$Hide :: @$Empty>
			variable "parentVertex" @Vertex
			parentVertex = Output#Sequence:
			parentVertex <+< @@Sequence ::> Output
			parentVertex -< @@Sequence ::> this\Output
		method "Dispose" ()
			<@System\Meta\Presentation\$Hide :: @$Empty>
			this\'AdditionalOctaveShift' = "0"
		method "Process" ()
			this.CopyAtomAttributes[]
			this.CopyEvents[]
			this.PutIntoStore[]
			this.Dispose[]
