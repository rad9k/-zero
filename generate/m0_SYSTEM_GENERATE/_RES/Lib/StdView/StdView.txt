		function "HandleGeneric_YOUTUBE" (@String "id")
			html += "<center><div class=%"video-container%"><div class=%"video-wrapper%"><iframe src=%"https://www.youtube.com/embed/"
			html += id
			html += "?rel=0&modestbranding=1&controls=1&showinfo=0&fs=1&autoplay=0%" allow=%"accelerometer; encrypted-media; gyroscope; picture-in-picture%" allowfullscreen></iframe></div></div></center>"
		function "HandleGeneric_AUDIO" (@String "url")
			html += "<audio class=%"player%" controls src=%""
			html += url
			html += "%"></audio>"
		function "HandleGeneric_DIAGRAM" (@String "Query", @String "label")
			html += "<center><img src=%"diagrams\"
			html += @@Html\DiagramQueryToDiagramId[Query]
			html += ".png%">"><BR><span class=%"description%">"
			html += label
			html += "</span></center>"
		function "MdTokenVertexesToHtmlString_Transform" (@Event "event", @VertexType "from", @VertexType "meta", @VertexType "to")
			variable "html" @String
			
			html = ""
			
			for edge "e" in from\
				test e\Meta
					case "Text"
						html += e\To
						
					case "HardBreak"
						html += "<BR>"
						
					case "ParagraphBreak"
						html += "<BR><BR>"
						
					case "HorizontalRule"
						html += "<HR>"
						
					case "Header1Start"
						html += "<H1>"
											
					case "Header2Start"
						html += "<H2>"
						
					case "Header3Start"
						html += "<H3>"
						
					case "Header4Start"
						html += "<H4>"
						
					case "Header5Start"
						html += "<H5>"
						
					case "Header6Start"
						html += "<H6>"
						
					case "Header1End"
						html += "</H1>"
											
					case "Header2End"
						html += "</H2>"
						
					case "Header3End"
						html += "</H3>"
						
					case "Header4End"
						html += "</H4>"
						
					case "Header5End"
						html += "</H5>"
						
					case "Header6End"
						html += "</H6>"
						
					case "BoldStart"
						html += "<B>"
						
					case "BoldEnd"
						html += "</B>"
						
					case "ItalicStart"
						html += "<I>"
						
					case "ItalicEnd"
						html += "</I>"
						
					case "CodeBlock"
						test e\To\FormalTextLanguageName == "InlineHTML"
							case "True"
								html += @@Html\DequoteText[e\To]
							case "False"							
								html += "<pre><code>"								
								test e\To\FormalTextLanguageName<> != "0"
									case "True"
										html += @@Html\AddColorsToCode[e\To\FormalTextLanguageName, e\To]
									case "False"
										html += e\To
								html += "</code></pre>"								
						
					case "InlineCodeStart"
						html += "<code>"
						
					case "InlineCodeEnd"
						html += "</code>"
						
					case "Link"
						html += "<a href=%""
						html += e\To\Url
						html += "%">"
						html += e\To\Text
						html += "</a>"
						
					case "Image"
						test e\To\Text<> == "0"
							case "True"
								html += "<center><img src=%""
								html += e\To\Url
								html += "%"></center>"
							case "False"
								html += "<center><img src=%""
								html += e\To\Url
								html += "%"><BR><span class=%"description%">"
								html += e\To\Text
								html += "</span></center>"
								
					case "Generic"
						test e\To
							case "YOUTUBE"
								@@Html\HandleGeneric_YOUTUBE[e\To\{Name:id}\Value]
							case "AUDIO"
								@@Html\HandleGeneric_AUDIO[e\To\{Name:url}\Value]
							case "DIAGRAM"
								@@Html\HandleGeneric_DIAGRAM[e\To\{Name:Query}\Value, e\To\{Name:label}\Value]
					
					case "BlockquoteStart"
						html += "<blockquote>"
						
					case "BlockquoteEnd"
						html += "</blockquote>"
						
					case "ListItemsStart"
						html += "<ul>"
						
					case "ListItemStart"
						html += "<li>"
						
					case "ListItemsEnd"
						html += "</ul>"
					
					case "OrderedListItemsStart"
						html += "<ol>"
						
					case "OrderedListItemStart"
						html += "<li>"
						
					case "OrderedListItemsEnd"
						html += "</ol>"
						
					case "TableStart"
						html += "<table>"
						
					case "HeaderBegin"
						html += "<thead>"
						
					case "HeaderEnd"
						html += "</thead>"
						
					case "TableEnd"
						html += "</table>"
						
					case "RowBegin"
						html += "<tr>"
						
					case "RowEnd"
						html += "</tr>"
						
					case "HeaderColumnBegin"
						html += "<td>"
						
					case "HeaderColumnEnd"
						html += "</td>"
						
					case "CellBegin"
						html += "<td>"
						
					case "CellEnd"
						html += "</td>"
						
					case "AlignLeft"
						html += "<div align=left>"
						
					case "AlignCenter"
						html += "<div align=center>"
						
					case "AlignRight"
						html += "<div align=right>"
						
				
			to <- @@Concatenate[html]
